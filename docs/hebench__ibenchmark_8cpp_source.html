50 std::unordered_set<std::size_t> PartialBenchmarkDescriptor::getCipherParamPositions(std::uint32_t cipher_param_mask)
51 {
52  std::unordered_set<std::size_t> retval;
53  std::bitset<sizeof(decltype(cipher_param_mask)) * 8> cipher_param_bits(cipher_param_mask);
54  for (std::size_t i = 0; i < cipher_param_bits.size(); ++i)
55  if (cipher_param_bits.test(i))
56  retval.insert(i);
57  return retval;
58 }

112 std::uint64_t PartialBenchmarkDescriptor::computeSampleSizes(std::uint64_t *sample_sizes,
113  std::size_t param_count,
114  const std::vector<std::uint64_t> &default_sample_sizes,
115  const hebench::APIBridge::BenchmarkDescriptor &bench_desc,
116  std::uint64_t default_sample_size_fallback)
117 {
118  if (bench_desc.category != hebench::APIBridge::Category::Offline)
119  throw std::invalid_argument(IL_LOG_MSG_CLASS("Invalid category. Only \"Offline\" category is supported."));
120  if (default_sample_size_fallback < 1)
121  throw std::invalid_argument(IL_LOG_MSG_CLASS("Default sample size fallback must be a positive value."));
122 
123  std::uint64_t result_batch_size = 1;
124  for (std::size_t param_i = 0; param_i < param_count; ++param_i)
125  {
126  std::uint64_t default_sample_size =
127  default_sample_sizes.size() > param_i && default_sample_sizes[param_i] > 0 ?
128  default_sample_sizes[param_i] :
129  default_sample_size_fallback;
130  sample_sizes[param_i] = (bench_desc.cat_params.offline.data_count[param_i] == 0 ?
131  default_sample_size :
132  bench_desc.cat_params.offline.data_count[param_i]);
133  result_batch_size *= sample_sizes[param_i];
134  } // end for
135  return result_batch_size;
136 }

138 IBenchmarkDescriptor::DescriptionToken::Ptr PartialBenchmarkDescriptor::matchDescriptor(const Engine &engine,
139  const BenchmarkDescription::Backend &backend_desc,
140  const BenchmarkDescription::Configuration &config) const
141 {
142  IBenchmarkDescriptor::DescriptionToken::Ptr retval;
143 
144  BenchmarkDescription::Backend final_backend_desc = backend_desc;
145  BenchmarkDescription::Configuration final_config = config;
146  const auto &w_params = final_config.w_params;
147  std::uint64_t w_params_count, other;
148  engine.validateRetCode(hebench::APIBridge::getWorkloadParamsDetails(engine.handle(), final_backend_desc.handle, &w_params_count, &other));
149  if (w_params_count != w_params.size())
150  {
151  std::stringstream ss;
152  ss << "Invalid number of workload arguments. Expected " << w_params_count << ", but " << w_params.size() << " received.";
153  throw std::runtime_error(IL_LOG_MSG_CLASS(ss.str()));
154  } // end if
155 
156  if (matchBenchmarkDescriptor(final_backend_desc.descriptor, w_params))
157  {
158  // complete the benchmark specification
159  BenchmarkDescription::Description text_description;
160  describe(engine, final_backend_desc, final_config, text_description);
161  retval = createToken(final_backend_desc, final_config, text_description);
162  } // end if
163 
164  return retval;
165 }

167 void PartialBenchmarkDescriptor::describe(const Engine &engine,
168  BenchmarkDescription::Backend &backend_desc,
169  BenchmarkDescription::Configuration &config,
170  BenchmarkDescription::Description &description) const
171 {
172  hebench::APIBridge::Handle h_bench_desc = backend_desc.handle;
173  const std::vector<hebench::APIBridge::WorkloadParam> &w_params = config.w_params;
174 
175  // obtain the values for the description that are specific to the workload
176  WorkloadDescriptionOutput completed_description;
177  std::memset(&completed_description.concrete_descriptor, 0, sizeof(completed_description.concrete_descriptor));
178  completeWorkloadDescription(completed_description,
179  engine, backend_desc, config);
180 
181  const hebench::APIBridge::BenchmarkDescriptor &bench_desc = completed_description.concrete_descriptor;
182 
183  std::stringstream ss;
184  std::filesystem::path ss_path;
185 
186  std::string &s_workload_name = completed_description.workload_name;
187  std::string s_path_workload_name;
188  std::string s_scheme_name = engine.getSchemeName(bench_desc.scheme);
189  std::string s_security_name = engine.getSecurityName(bench_desc.scheme, bench_desc.security);
190 
191  // generate path
192  ss = std::stringstream();
193  if (!s_workload_name.empty())
194  {
195  ss << s_workload_name << "_";
196  }
197  else
198  {
199  s_workload_name = std::to_string(static_cast<int>(bench_desc.workload));
200  }
201  ss << std::to_string(static_cast<int>(bench_desc.workload));
202  s_path_workload_name = hebench::Utilities::convertToDirectoryName(ss.str());
203  ss_path = s_path_workload_name;
204  ss = std::stringstream();
205  ss << "wp";
206  for (std::size_t i = 0; i < w_params.size(); ++i)
207  {
208  ss << "_";
209  switch (w_params[i].data_type)
210  {
211  case hebench::APIBridge::WorkloadParamType::UInt64:
212  ss << w_params[i].u_param;
213  break;
214 
215  case hebench::APIBridge::WorkloadParamType::Float64:
216  ss << w_params[i].f_param;
217  break;
218 
219  default:
220  ss << w_params[i].i_param;
221  break;
222  } // end switch
223  } // end for
224  ss_path /= hebench::Utilities::convertToDirectoryName(ss.str()); // workload params
225  ss_path /= hebench::Utilities::convertToDirectoryName(PartialBenchmarkDescriptor::getCategoryName(bench_desc.category));
226  ss_path /= hebench::Utilities::convertToDirectoryName(PartialBenchmarkDescriptor::getDataTypeName(bench_desc.data_type));
227 
228  ss = std::stringstream();
229  ss << bench_desc.cat_params.min_test_time_ms << "ms";
230  std::size_t max_non_zero = HEBENCH_MAX_CATEGORY_PARAMS;
231  while (max_non_zero > 0 && bench_desc.cat_params.reserved[max_non_zero - 1] == 0)
232  --max_non_zero;
233  if (max_non_zero > 0)
234  {
235  for (std::size_t i = 0; i < max_non_zero; ++i)
236  ss << "_" << bench_desc.cat_params.reserved[i];
237  } // end if
238  else
239  ss << "_default";
240 
241  ss_path /= ss.str();
242  // cipher/plain parameters
243  ss = std::stringstream();
244  std::unordered_set<std::size_t> cipher_param_pos =
245  PartialBenchmarkDescriptor::getCipherParamPositions(bench_desc.cipher_param_mask);
246  if (cipher_param_pos.empty())
247  ss << "all_plain";
248  else if (cipher_param_pos.size() >= sizeof(std::uint32_t) * 8)
249  ss << "all_cipher";
250  else
251  {
252  std::size_t max_elem = *std::max_element(cipher_param_pos.begin(), cipher_param_pos.end());
253  for (std::size_t i = 0; i <= max_elem; ++i)
254  ss << (cipher_param_pos.count(i) > 0 ? 'c' : 'p');
255  } // end else
256  ss_path /= ss.str();
257  ss_path /= hebench::Utilities::convertToDirectoryName(s_scheme_name);
258  ss_path /= hebench::Utilities::convertToDirectoryName(s_security_name);
259  ss_path /= std::to_string(bench_desc.other);
260  // generate header
261  ss = std::stringstream();
262  ss << "Specifications," << std::endl
263  << ", Encryption, " << std::endl
264  << ", , Scheme, " << s_scheme_name << std::endl
265  << ", , Security, " << s_security_name << std::endl
266  << ", Extra, " << bench_desc.other << std::endl;
267  std::string s_tmp = engine.getExtraDescription(h_bench_desc, w_params);
268  if (!s_tmp.empty())
269  ss << s_tmp;
270  ss << std::endl
271  << std::endl
272  << ", Category, " << PartialBenchmarkDescriptor::getCategoryName(bench_desc.category) << std::endl
273  << ", , Minimum test time requested (ms), " << bench_desc.cat_params.min_test_time_ms << std::endl;
274  switch (bench_desc.category)
275  {
276  case hebench::APIBridge::Category::Latency:
277  ss << ", , Warmup iterations, " << bench_desc.cat_params.latency.warmup_iterations_count << std::endl;
278  break;
279 
280  case hebench::APIBridge::Category::Offline:
281  {
282  ss << ", , Parameter, Samples requested" << std::endl;
283 
284  bool all_params_zero = true;
285  for (std::size_t i = 0; i < HEBENCH_MAX_OP_PARAMS; ++i)
286  {
287  if (bench_desc.cat_params.offline.data_count[i] != 0)
288  {
289  all_params_zero = false;
290  ss << ", , " << i << ", " << bench_desc.cat_params.offline.data_count[i] << std::endl;
291  } // end if
292  } // end if
293  if (all_params_zero)
294  ss << ", , All, 0" << std::endl;
295  }
296  break;
297 
298  default:
299  throw std::invalid_argument(IL_LOG_MSG_CLASS("Unsupported benchmark category: " + std::to_string(bench_desc.category) + "."));
300  break;
301  } // end switch
302 
303  ss << std::endl
304  << ", Workload, " << s_workload_name << std::endl
305  << ", , Data type, " << PartialBenchmarkDescriptor::getDataTypeName(bench_desc.data_type) << std::endl
306  << ", , Encrypted op parameters (index)";
307  if (cipher_param_pos.empty())
308  ss << ", None" << std::endl;
309  else if (cipher_param_pos.size() >= sizeof(std::uint32_t) * 8)
310  ss << ", All" << std::endl;
311  else
312  {
313  // output the indices of the encrypted parameters
314  std::vector<std::size_t> cipher_params(cipher_param_pos.begin(), cipher_param_pos.end());
315  std::sort(cipher_params.begin(), cipher_params.end());
316  for (auto param_index : cipher_params)
317  ss << ", " << param_index;
318  ss << std::endl;
319  } // end else
320 
321  // append the extra header received from the workload
322  ss << completed_description.workload_header;
323 
324  // return the generated description
325 
326  // make sure we have default sample sizes for every operation parameter
327  Engine::completeBenchmarkDescriptor(backend_desc, completed_description.concrete_descriptor);
328  backend_desc.operation_params_count = completed_description.operation_params_count;
329  config.default_sample_sizes.resize(backend_desc.operation_params_count, 0);
330 
331  description.workload_name = completed_description.workload_name;
332  description.header = ss.str();
333  description.path = ss_path;
334  if (config.b_single_path_report)
335  // replace path separator by hyphens when requested
336  std::replace(description.path.begin(), description.path.end(),
337  hebench::TestHarness::separator, hebench::TestHarness::hyphen);
338 }

382 void PartialBenchmark::initBackend(hebench::Utilities::TimingReportEx &out_report, const FriendPrivateKey &)
383 {
384  hebench::Common::EventTimer timer;
385  hebench::Common::TimingReportEvent::Ptr p_timing_event;
386  std::stringstream ss;
387 
388  hebench::APIBridge::WorkloadParams params;
389  params.count = m_config.w_params.size();
390  params.params = m_config.w_params.data();
391 
392  std::cout << IOS_MSG_INFO << hebench::Logging::GlobalLogger::log("Creating backend benchmark...") << std::endl;
393  timer.start();
394  validateRetCode(hebench::APIBridge::createBenchmark(m_p_engine->handle(),
395  this->getBackendDescription().handle,
396  m_config.w_params.empty() ? nullptr : &params,
397  &m_handle));
398  p_timing_event = timer.stop<DefaultTimeInterval>(getEventIDNext(), 1, nullptr);
399  out_report.addEvent<DefaultTimeInterval>(p_timing_event, std::string("Creation"));
400  hebench::Logging::GlobalLogger::log("OK");
401  std::cout << IOS_MSG_OK << std::endl;
402 
403  std::cout << IOS_MSG_INFO << hebench::Logging::GlobalLogger::log("Initializing backend benchmark...") << std::endl;
404  timer.start();
405  validateRetCode(hebench::APIBridge::initBenchmark(m_handle, &this->getBackendDescription().descriptor));
406  p_timing_event = timer.stop<DefaultTimeInterval>(getEventIDNext(), 1, nullptr);
407  out_report.addEvent<DefaultTimeInterval>(p_timing_event, std::string("Initialization"));
408  hebench::Logging::GlobalLogger::log("OK");
409  std::cout << IOS_MSG_OK << std::endl;
410 }
