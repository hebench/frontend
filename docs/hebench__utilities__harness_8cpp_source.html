25  [to_lowercase](unsigned char c) { return (std::isalnum(c) || c == '.' ?
26  (to_lowercase ? std::tolower(c) : c) :
27  '_'); });

29  s_tmp.erase(0, s_tmp.find_first_not_of('_'));

35  {
36  std::size_t i = s_tmp.find_first_of('_', pos);

45  } // end if
